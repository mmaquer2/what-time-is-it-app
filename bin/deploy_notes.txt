# ssh to the server

sudo apt-get update

# install nginx
sudo apt install nginx


# verify ngnix 
sudo nginx -t


# enable ngnix on boot
sudo systemctl start nginx
sudo systemctl enable nginx

# install nodejs and npm
sudo apt install npm
sudo apt install -y nodejs

# verify installation 
node -v
npm -v
